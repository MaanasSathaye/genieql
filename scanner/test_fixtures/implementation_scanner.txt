type Scanner struct {
	Rows *sql.Rows
}

func (t Scanner) Scan(arg0 *CustomType) error {
	var c1 bool
	var c2 sql.NullBool
	if err := t.Rows.Scan(&c1, &c2); err != nil {
		return err
	}
	arg0.Field1 = c1
	if c2.Valid {
		tmp := c2.Bool
		arg0.Field2 = &tmp
	}
	return t.Rows.Err()
}
func (t Scanner) Err() error {
	return t.Rows.Err()
}
func (t Scanner) Close() error {
	if t.Rows == nil {
		return nil
	}
	return t.Rows.Close()
}
func (t Scanner) Next() bool {
	return t.Rows.Next()
}
